Group By

Groups the elements of an observable sequence according to a specified key selector function and comparer and selects the resulting elements by using a specified function.

The example below shows a list of loans given to multiple people. As part of reporting you might want to know the total loan amount for all the different term. for example, what's the total loan amount for all loans where the term is 12 months? 

var codes = [
    { term: 12, loan: 1000},
    { term: 12, loan: 1000},
    { term: 24, loan: 1000},
    { term: 60, loan: 1000},
    { term: 60, loan: 1000},
    { term: 60, loan: 1000},
    { term: 120, loan: 1000}
];

var source = Rx.Observable
    .from(codes)
    .groupBy(
        function (x) { return x.term; },
        function (x) { return x.loan; }
    );
        
var subscription = source.subscribe(
    function (obs) {
        obs
            .sum()
            .subscribe(function (x) {
                console.log('Total Loans ' + x + ' of Term ' + obs.key);
        });
    },
    function (err) {
        console.log('Error: ' + err);
    },
    function () {
        console.log('Completed');
    });
